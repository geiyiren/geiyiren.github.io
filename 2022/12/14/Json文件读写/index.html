java读写Json文件

json格式介绍

json与xml相比, 对数据的描述性比XML较差,但是数据体积小,传递速度更快.

json数据的书写格式是”名称:值对”,比如:

"Name" : "John"                        //name为名称,值对为"john"字符串

值对类型共分为:

  • 数字（整数或浮点数）
  • 字符串（在双引号中）
  • 逻辑值（true 或 false）
  • 数组（在方括号[]中）
  • 对象（在花括号{}中）
  • null
    当然数组也可以包含多对象:
{
    "employees": [
        { "Name":"John" , "Age":19 },
        { "Name":"Anna" , "Age":22 },
        { "Name":"Peter", "Age":23 }
    ]
}

表示”employees”对象中有3个对象数组(每个对象数组表示一条员工信息),其中并列的数据都必须用逗号”,”隔开.

json文件读写示例

package com.fibocom.multicamera;

import android.util.Log;

import org.json.JSONArray;
import org.json.JSONException;
import org.json.JSONObject;

import java.io.File;
import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.FileOutputStream;
import java.io.InputStreamReader;
import java.io.OutputStreamWriter;
import java.io.UnsupportedEncodingException;

class carParameter  extends Object{
    public int on_TrajectoryLine;
    public int on_BSD;
    public int on_Bright;
    public double scale_size; // 缩放
    public int car_length;//车辆长度
    public int car_width;//车辆宽度
    public int suspension_front;//前悬挂
    public int suspension_back;//后悬挂
}

public class AvmSetParm {
    public String jsonFileName;
    public carParameter readJson() throws Exception {
            carParameter carParm=new carParameter();
            char cbuf[] = new char[10000];
            InputStreamReader input =new InputStreamReader(new FileInputStream(new File(jsonFileName)),"UTF-8");
            int len =input.read(cbuf);
            String text =new String(cbuf,0,len);
            //1.构造一个json对象
            JSONObject obj = new JSONObject(text.substring(text.indexOf("{")));   //过滤读出的utf-8前三个标签字节,从{开始读取

            //获取数组
            JSONArray dataArr = obj.getJSONArray("data");
            JSONObject carParameterObj =dataArr.getJSONObject(3);
            JSONObject carParameter=carParameterObj.getJSONObject("carParameter");

            carParm.car_length=carParameter.getInt("car_length");
            carParm.car_width=carParameter.getInt("car_width");
            carParm.suspension_front=carParameter.getInt("suspension_front");
            carParm.suspension_back=carParameter.getInt("suspension_back");
            carParm.scale_size=carParameter.getDouble("scale_size");
            carParm.on_Bright=carParameter.getInt("BE_Switch");
            carParm.on_BSD=carParameter.getInt("BSD_Switch");
            carParm.on_TrajectoryLine=carParameter.getInt("TrajectoryLine_Switch");
            input.close();

            Log.d("AvmSetParm","car_length:"+carParm.car_length);
            Log.d("AvmSetParm","car_width:"+carParm.car_width);
            Log.d("AvmSetParm","suspension_front:"+carParm.suspension_front);
            Log.d("AvmSetParm","suspension_back:"+carParm.suspension_back);
            Log.d("AvmSetParm","scale_size:"+carParm.scale_size);
            return carParm;
    }
    public void writeJson(carParameter carParm) throws Exception {
        char cbuf[] = new char[10000];
        InputStreamReader input =new InputStreamReader(new FileInputStream(new File(jsonFileName)),"UTF-8");

        int len =input.read(cbuf);
        String text =new String(cbuf,0,len);
        //1.构造一个json对象
        JSONObject obj = new JSONObject(text.substring(text.indexOf("{")));   //过滤读出的utf-8前三个标签字节,从{开始读取

        //获取数组
        JSONArray data = obj.getJSONArray("data");
        JSONObject carParameterObj =data.getJSONObject(3);
        JSONObject carParameter=carParameterObj.getJSONObject("carParameter");

        carParameter.put("car_length",carParm.car_length);
        carParameter.put("car_width",carParm.car_width);
        carParameter.put("suspension_front",carParm.suspension_front);
        carParameter.put("suspension_back",carParm.suspension_back);
        carParameter.put("BE_Switch",carParm.on_Bright);
        carParameter.put("BSD_Switch",carParm.on_BSD);
        carParameter.put("TrajectoryLine_Switch",carParm.on_TrajectoryLine);
        Log.d("AvmSetParm", "car_length"+carParm.car_length);
        Log.d("AvmSetParm", "car_width"+carParm.car_width);
        Log.d("AvmSetParm", "suspension_front"+carParm.suspension_front);
        Log.d("AvmSetParm", "suspension_back"+carParm.suspension_back);
        Log.d("AvmSetParm", "BE_Switch"+carParm.on_Bright);
        Log.d("AvmSetParm", "BSD_Switch"+carParm.on_BSD);
        Log.d("AvmSetParm", "TrajectoryLine_Switch"+carParm.on_TrajectoryLine);

        carParameterObj.put("carParameter",carParameter);
        data.put(3,carParameterObj);
        obj.put("data",data);

        OutputStreamWriter osw = new OutputStreamWriter( new FileOutputStream(jsonFileName),"UTF-8");
        osw.write(obj.toString(4));
        osw.flush();//清空缓冲区，强制输出数据
        osw.close();//关闭输出流
    }
}

  • 模板函数定义
/**
 * @brief 从文本中加载json
 * @tparam R json类型
 * @tparam T 自动推断的字符串类型
 * @param config json字符串
 * @return R
 */
template<typename R, typename T>
inline R LoadJsonFromText(const T &config)
{
    // static_assert(JsonHasParse<R>::value,
    //               "error: the type must has static method [parse].");
    // nlohmann json
    return R::parse(config, nullptr, true, true);
}

/**
 * @brief 通用类型
 * @tparam R
 * @tparam T
 * @tparam U
 * @param config
 * @param key
 * @param val
 * @return R
 */
template<typename R, typename T, typename U>
inline R OptionalContain(const T &config, const U &key, const R &val, JsonGenericType)
{
    if (config.count(key) > 0)
    {
        return config.at(key);
    }
    return val;
}
