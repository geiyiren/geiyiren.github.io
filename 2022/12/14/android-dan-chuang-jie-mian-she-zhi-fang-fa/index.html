android弹窗界面设置方法

android 安卓弹窗
android

更新日期:   2025-08-14

功能描述

点击主界面的一个按钮，然后跳转另一个设置界面，进行参数设置，包括：参数输入、开关按钮、保存、退出等功能。界面的布局采用表格布局，平均分布各个行列。示例代码如下所示。

MainActivity.java示例

// 主界面按钮定义
private ImageButton mImage_set_btn;

@Override
protected void onCreate(Bundle savedInstanceState) {
	super.onCreate(savedInstanceState);

	// 主界面按钮事件
	// R.id.image_set_btn为主界面按钮ID
	mImage_set_btn = (ImageButton) findViewById(R.id.image_set_btn);
	mImage_set_btn.setOnClickListener(new View.OnClickListener() {
	@Override
	public void onClick(View view) {
		final LayoutInflater layoutInflater = (LayoutInflater)getBaseContext().getSystemService(LAYOUT_INFLATER_SERVICE);
		//R.id.avm_parm_view为跳转界面ID（android:id="@+id/avm_parm_view"）
		View popupView = layoutInflater.inflate(R.layout.layout_avm_parm, (ViewGroup)findViewById(R.id.avm_parm_view));
		mPopupWindow_register = new PopupWindow(popupView, WindowManager.LayoutParams.MATCH_PARENT, WindowManager.LayoutParams.MATCH_PARENT, true);
		mPopupWindow_register.setTouchable(true);
		mPopupWindow_register.setOutsideTouchable(true);
		mPopupWindow_register.setBackgroundDrawable(new BitmapDrawable(getResources(), (Bitmap) null));
		mPopupWindow_register.showAsDropDown(view);

		//------ 获取输入控件的输入内容 ------
		//车长
		editText_car_length = (EditText) mPopupWindow_register.getContentView().findViewById(R.id.EditText_car_length);
		//车宽
		editText_car_width = (EditText) mPopupWindow_register.getContentView().findViewById(R.id.EditText_car_width);
		//前悬
		editText_suspension_front = (EditText) mPopupWindow_register.getContentView().findViewById(R.id.EditText_suspension_front);
		//后悬
		editText_suspension_back = (EditText) mPopupWindow_register.getContentView().findViewById(R.id.EditText_suspension_back);
		//保存按钮
		delok_btn = popupView.findViewById(R.id.avm_parameter_save);
		//返回按钮
		delcancel_btn = popupView.findViewById(R.id.avm_parameter_return);
		//车道线开关
		mTrajectory_Switch=popupView.findViewById(R.id.TrajectoryLine_Switch);
		//BSD开关
		mBSD_Switch=popupView.findViewById(R.id.BSD_Switch);
		//亮度均衡开关
		mBE_Switch=popupView.findViewById(R.id.BE_Switch);
		// 获取默认开关参数
		carParameter carParm_history = new carParameter();
		try {
			carParm_history=avmSetParm.readJson();
		} catch (Exception e) {
			e.printStackTrace();
		}
		//------ 设置输入控件的显示历史数据 ------
		editText_car_length.setText(String.valueOf(carParm_history.car_length));
		editText_car_width.setText(String.valueOf(carParm_history.car_width));
		editText_suspension_front.setText(String.valueOf(carParm_history.suspension_front));
		editText_suspension_back.setText(String.valueOf(carParm_history.suspension_back));
		switch (carParm_history.on_TrajectoryLine){
			case 0:
				mTrajectory_Switch.setChecked(false);
				break;
			case 1:
				mTrajectory_Switch.setChecked(true);
				break;
		}
		switch (carParm_history.on_BSD){
			case 0:
				mBSD_Switch.setChecked(false);
				break;
			case 1:
				mBSD_Switch.setChecked(true);
				break;
		}
		switch (carParm_history.on_Bright){
			case 0:
				mBE_Switch.setChecked(false);
				break;
			case 1:
				mBE_Switch.setChecked(true);
				break;
		}

		//------ 设置保存按钮触发事件 ------
		delok_btn.setOnClickListener(new View.OnClickListener() {
			public void onClick(View v) {
				String car_length = editText_car_length.getEditableText().toString();
				String car_width = editText_car_width.getEditableText().toString();
				String suspension_front = editText_suspension_front.getEditableText().toString();
				String suspension_back = editText_suspension_back.getEditableText().toString();


				carParameter carParm = new carParameter();

				Log.d("filename_avm_set_json","filename_avm_set_json"+avmSetParm.jsonFileName);
				carParm.car_length=Integer.parseInt(car_length);
				carParm.car_width=Integer.parseInt(car_width);
				carParm.suspension_front=Integer.parseInt(suspension_front);
				carParm.suspension_back=Integer.parseInt(suspension_back);
				carParm.on_BSD=mView.mBSDViewID;
				carParm.on_Bright=mView.mBrightID;
				carParm.on_TrajectoryLine=mView.mTrajectoryLineViewID;
				Log.d(TAG, "car_length"+carParm.car_length);
				Log.d(TAG, "car_width"+carParm.car_width);
				Log.d(TAG, "suspension_front"+carParm.suspension_front);
				Log.d(TAG, "suspension_back"+carParm.suspension_back);
				Log.d(TAG, "on_BSD"+carParm.on_BSD);
				Log.d(TAG, "on_Bright"+carParm.on_Bright);
				Log.d(TAG, "on_TrajectoryLine"+carParm.on_TrajectoryLine);
				try {
					avmSetParm.writeJson(carParm);
				} catch (Exception e) {
					e.printStackTrace();
				}
				mPopupWindow_register.dismiss();
			}
		});

		//------ 设置按钮触发事件 ------
		//车道线
		mTrajectory_Switch.setOnCheckedChangeListener(new CompoundButton.OnCheckedChangeListener() {
			@Override
			public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
				if (buttonView.isChecked()){
					mView.mTrajectoryLineViewID=1;
				}else {
					mView.mTrajectoryLineViewID=0;
				}
				mView.avmTrajectoryLineView();
			}
		});

		//BSD
		mBSD_Switch.setOnCheckedChangeListener(new CompoundButton.OnCheckedChangeListener() {
			@Override
			public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
				if (buttonView.isChecked()){
					mView.mBSDViewID=1;
					avmStartBSD();
				}else {
					mView.mBSDViewID=0;
					avmStopBSD();
				}
				mView.avmBSDView();
			}
		});
		//亮度均衡
		mBE_Switch.setOnCheckedChangeListener(new CompoundButton.OnCheckedChangeListener() {
			@Override
			public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
				if (buttonView.isChecked()){
					mView.mBrightID=1;
				}else {
					mView.mBrightID=0;
				}
				mView.avmBrightView();
			}
		});
		//返回
		delcancel_btn.setOnClickListener(new View.OnClickListener() {
			public void onClick(View v) {
				mPopupWindow_register.dismiss();
			}
		});
	}
});

}
