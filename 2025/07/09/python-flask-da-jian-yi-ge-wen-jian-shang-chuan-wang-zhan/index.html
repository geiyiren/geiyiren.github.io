python+flask搭建一个文件上传网站

python flask

用 Python + Flask 构建的简洁网页上传工程，包含一下功能：
1.误报视频上传
2.json配置文件上传
3.误报类型单选，仅支持：fcw,ldw,dsm
4.误报类型选择dsm外其他类型时，json配置文件为必须上传到文件
5.误报说明文字框
6.项目名称文字框
7.历史提交记录展示窗口，以上传填写的内容为表头展示列表。，增加解决按钮，点击此按钮需要填写密码101728
8.上传人填写
9.提交按钮
10.每次上传的所有文件以时间戳命名存在单独的一个文件夹中
11.给出工程的目录结构创建的shell脚本
12.网页的ip为172.20.1.108，端口8080

工程创建脚本

#!/bin/bash

mkdir -p templates
mkdir -p uploads
mkdir -p logs

touch app.py
touch templates/index.html
touch logs/submissions.log

echo "目录结构创建完成。"

app.py

import os
import time
import json
from flask import Flask, render_template, request, redirect, url_for, flash, jsonify

app = Flask(__name__)
app.secret_key = 'some_secret_key_for_flash'

UPLOAD_FOLDER = 'uploads'
HISTORY_FILE = 'history.json'

# 允许的误报类型
ALLOWED_TYPES = ['fcw', 'ldw', 'dsm']

# 确保上传文件夹存在
os.makedirs(UPLOAD_FOLDER, exist_ok=True)

def load_history():
    if os.path.exists(HISTORY_FILE):
        with open(HISTORY_FILE, 'r', encoding='utf-8') as f:
            try:
                return json.load(f)
            except:
                return []
    else:
        return []

def save_history(history):
    with open(HISTORY_FILE, 'w', encoding='utf-8') as f:
        json.dump(history, f, indent=2, ensure_ascii=False)

@app.route('/', methods=['GET', 'POST'])
def index():
    history = load_history()
    if request.method == 'POST':
        # 获取表单数据
        uploader = request.form.get('uploader', '').strip()
        project_name = request.form.get('project_name', '').strip()
        report_type = request.form.get('report_type')
        description = request.form.get('description', '').strip()
        password = request.form.get('password', '').strip()
        solve_index = request.form.get('solve_index')

        # 解决按钮提交处理（带密码）
        if solve_index is not None:
            if password == '101728':
                history = load_history()
                idx = int(solve_index)
                if 0 <= idx < len(history):
                    history[idx]['solved'] = True
                    save_history(history)
                    flash('问题已标记为已解决', 'success')
                else:
                    flash('索引错误', 'danger')
            else:
                flash('密码错误', 'danger')
            return redirect(url_for('index'))

        # 上传文件
        video_file = request.files.get('video_file')
        json_file = request.files.get('json_file')

        # 验证上传内容
        if report_type not in ALLOWED_TYPES:
            flash('请选择正确的误报类型', 'danger')
            return redirect(url_for('index'))

        if not video_file or video_file.filename == '':
            flash('必须上传误报视频文件', 'danger')
            return redirect(url_for('index'))

        # 如果类型不是 dsm，json文件为必须上传
        if report_type != 'dsm':
            if not json_file or json_file.filename == '':
                flash('误报类型非dsm时，必须上传JSON配置文件', 'danger')
                return redirect(url_for('index'))

        # 创建时间戳文件夹
        timestamp = time.strftime('%Y%m%d_%H%M%S')
        save_dir = os.path.join(UPLOAD_FOLDER, timestamp)
        os.makedirs(save_dir, exist_ok=True)

        # 保存视频文件
        video_filename = f'video_{int(time.time())}_{video_file.filename}'
        video_path = os.path.join(save_dir, video_filename)
        video_file.save(video_path)

        json_path = None
        if json_file and json_file.filename != '':
            json_filename = f'config_{int(time.time())}_{json_file.filename}'
            json_path = os.path.join(save_dir, json_filename)
            json_file.save(json_path)

        # 更新历史记录
        history = load_history()
        new_record = {
            'timestamp': timestamp,
            'project_name': project_name,
            'report_type': report_type,
            'description': description,
            'uploader': uploader,
            'video_file': video_filename,
            'json_file': json_file.filename if json_file else '',
            'solved': False
        }
        history.insert(0, new_record)  # 最新在前
        save_history(history)

        flash('提交成功', 'success')
        return redirect(url_for('index'))

    return render_template('index.html', history=load_history(), allowed_types=ALLOWED_TYPES)
